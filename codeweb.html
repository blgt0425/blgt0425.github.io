Rock Paper Scissors - C++ Code

#include <iostream>
#include <ctime>
#include <string>
using namespace std;

int choice;
void computer();
void player();
string r = "rock";
string p = "paper";
string s = "scissors";

int main(){
    srand(time(0));
    cout << "Rock Paper Scissors\n";
    cout << "1 = Rock\n2 = Paper\n3 = Scissors\n";
    player();
    computer();
}

void player(){
    do{
        cin >> choice;
        switch(choice){
            case 1:
                cout << "\nPLAYER'S CHOICE\nP1 has chosen " << r << endl;
                cout << "====================\n";
                break;
            case 2:
                cout << "\nPLAYER'S CHOICE\nP1 has chosen " << p << endl;
                cout << "====================\n";
                break;
            case 3:
                cout << "\nPLAYER'S CHOICE\nP1 has chosen " << s << endl;
                cout << "====================\n";
                break;
            default:
                cout << "\nPLAYER'S CHOICE\nInvalid choice\n====================\n";
        }
    } while(choice > 3);
}

void computer(){
    int randnum = rand() % 3 + 1;
    switch(randnum){
        case 1:
            cout << "COMPUTER'S CHOICE\nP2 has chosen " << r << endl;
            cout << "====================\n";
            break;
        case 2:
            cout << "COMPUTER'S CHOICE\nP2 has chosen " << p << endl;
            cout << "====================\n";
            break;
        case 3:
            cout << "COMPUTER'S CHOICE\nP2 has chosen " << s << endl;
            cout << "====================\n";
            break;
    }

    if((randnum == 1 && choice == 2) || 
       (randnum == 2 && choice == 3) || 
       (randnum == 3 && choice == 1)){
        cout << "RESULT\n==============\nYou won\n";
    } else if((randnum == 1 && choice == 3) || 
              (randnum == 2 && choice == 1) || 
              (randnum == 3 && choice == 2)){
        cout << "RESULT\n==============\nComputer won\n";
    } else {
        cout << "RESULT\n==============\nTie\n";
    }
}
